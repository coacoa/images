HTML / CSS / JS を左右分割で理解

家の骨組み

HTML 
<div id="house" class="house">
  <!-- 屋根（CSSで三角形） -->
  <div class="roof"></div>

  <!-- 壁（普通の四角形） -->
  <div class="walls">
    <p>家の骨組み</p>
    <div id="door" class="door"></div>
  </div>
</div>
CSS 
/* ▼ 三角形の屋根 */
.roof {
  width: 0;
  height: 0;
  border-left: 130px solid transparent;
  border-right: 130px solid transparent;
  border-bottom: 80px solid #bbb;
}

/* ▼ 四角形の壁 */
.walls {
  padding: 20px;
  background: white;
}

/* 装飾ON時 */
.decorated .roof { border-bottom-color: orange; }
.decorated .walls { background: #e3f2fd; }
.decorated .door { background: #795548; }
JavaScript 
function toggleCSS() {
  house.classList.toggle('decorated');
}

function toggleDoor() {
  door.classList.toggle('door-open');
}
